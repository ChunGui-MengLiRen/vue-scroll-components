# 使用

## 全局使用
```js
// main.js
import { createApp } from "vue"
import vueScrollComponentfrom "vue-scroll-components"
import App from "./App.vue"
const app = createApp(App)
app.use(vueScrollComponentfrom)
app.mount("#app");
```
